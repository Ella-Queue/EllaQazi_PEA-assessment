CREATE TABLE AdditionalItems
(
  ItemId INT NOT NULL,
  ItemDescription VARCHAR(4) NOT NULL,
  PRIMARY KEY (ItemId)
);

CREATE TABLE EmergencyContact
(
  EmergencyContactID INT NOT NULL,
  FirstName VARCHAR(20) NOT NULL,
  LastName VARCHAR(20) NOT NULL,
  Phone INT NOT NULL,
  PRIMARY KEY (EmergencyContactID)
);

CREATE TABLE Account
(
  AccountId INT NOT NULL,
  Username VARCHAR(30) NOT NULL,
  FirstName VARCHAR(20) NOT NULL,
  LastName VARCHAR(20) NOT NULL,
  Type VARCHAR(8) NOT NULL,
  EmergencyContactID INT NOT NULL,
  PRIMARY KEY (AccountId),
  FOREIGN KEY (EmergencyContactID) REFERENCES EmergencyContact(EmergencyContactID)
);

CREATE TABLE Customer
(
  Address VARCHAR(40) NOT NULL,
  City VARCHAR(20) NOT NULL,
  PostalCode CHAR(6) NOT NULL,
  Province VARCHAR(20) NOT NULL,
  HomePhone INT NOT NULL,
  CellPhone INT NOT NULL,
  Email VARCHAR(40) NOT NULL,
  Password VARCHAR(30) NOT NULL,
  AccountId INT NOT NULL,
  PRIMARY KEY (AccountId),
  FOREIGN KEY (AccountId) REFERENCES Account(AccountId)
);

CREATE TABLE Reservation
(
  ReservationId INT NOT NULL,
  StartDate DATE NOT NULL,
  EndDate DATE NOT NULL,
  PickupTime CHAR(5) NOT NULL,
  PRIMARY KEY (ReservationId)
);

CREATE TABLE Invoice
(
  ReservationId INT NOT NULL,
  PRIMARY KEY (ReservationId),
  FOREIGN KEY (ReservationId) REFERENCES Reservation(ReservationId)
);

CREATE TABLE Food
(
  FoodId INT NOT NULL,
  Type CHAR(3) NOT NULL,
  Brand VARCHAR(30) NOT NULL,
  Name VARCHAR(60) NOT NULL,
  PRIMARY KEY (FoodId)
);

CREATE TABLE Pet
(
  PetId INT NOT NULL,
  Name VARCHAR(20) NOT NULL,
  Species CHAR(3) NOT NULL,
  Sterelized NOT NULL,
  SpecialNotes VARCHAR NOT NULL,
  Sex CHAR(1) NOT NULL,
  BirthDate DATE NOT NULL,
  Breed VARCHAR(20) NOT NULL,
  AccountId INT NOT NULL,
  PRIMARY KEY (PetId),
  FOREIGN KEY (AccountId) REFERENCES Customer(AccountId)
);

CREATE TABLE Housing
(
  HousingId INT NOT NULL,
  Type VARCHAR(5) NOT NULL,
  PRIMARY KEY (HousingId)
);

CREATE TABLE PetReservation
(
  PetReservationID INT NOT NULL,
  ReservationId INT NOT NULL,
  FoodId INT NOT NULL,
  PetId INT NOT NULL,
  HousingId INT NOT NULL,
  PRIMARY KEY (ReservationId, PetId, PetReservationID),
  FOREIGN KEY (ReservationId) REFERENCES Reservation(ReservationId),
  FOREIGN KEY (FoodId) REFERENCES Food(FoodId),
  FOREIGN KEY (PetId) REFERENCES Pet(PetId),
  FOREIGN KEY (HousingId) REFERENCES Housing(HousingId)
);

CREATE TABLE Vaccination
(
  VaccinationId INT NOT NULL,
  Name VARCHAR(20) NOT NULL,
  Type CHAR(3) NOT NULL,
  PRIMARY KEY (VaccinationId)
);

CREATE TABLE Employee
(
  AccountId INT NOT NULL,
  PRIMARY KEY (AccountId),
  FOREIGN KEY (AccountId) REFERENCES Account(AccountId)
);

CREATE TABLE Owner
(
  AccountId INT NOT NULL,
  PRIMARY KEY (AccountId),
  FOREIGN KEY (AccountId) REFERENCES Account(AccountId)
);

CREATE TABLE Dog
(
  Size CHAR(1) NOT NULL,
  Climber 1 NOT NULL,
  Barker 1 NOT NULL,
  PetId INT NOT NULL,
  PRIMARY KEY (PetId),
  FOREIGN KEY (PetId) REFERENCES Pet(PetId)
);

CREATE TABLE Cat
(
  Type VARCHAR(7) NOT NULL,
  LitterTrained 1 NOT NULL,
  PetId INT NOT NULL,
  PRIMARY KEY (PetId),
  FOREIGN KEY (PetId) REFERENCES Pet(PetId)
);

CREATE TABLE Run
(
  Size CHAR(1) NOT NULL,
  HousingId INT NOT NULL,
  PRIMARY KEY (HousingId),
  FOREIGN KEY (HousingId) REFERENCES Housing(HousingId)
);

CREATE TABLE Suite
(
  Accomodation VARCHAR(11) NOT NULL,
  HousingId INT NOT NULL,
  PRIMARY KEY (HousingId),
  FOREIGN KEY (HousingId) REFERENCES Housing(HousingId)
);

CREATE TABLE Service
(
  ServiceId INT NOT NULL,
  ServiceName CHAR(1) NOT NULL,
  PRIMARY KEY (ServiceId)
);

CREATE TABLE Price
(
  PriceID INT NOT NULL,
  PriceName VARCHAR(20) NOT NULL,
  PriceCost NOT NULL,
  PRIMARY KEY (PriceID)
);

CREATE TABLE Medication
(
  MedicationId INT NOT NULL,
  Type VARCHAR(80) NOT NULL,
  Frequency INT NOT NULL,
  Dosage INT NOT NULL,
  ReservationId INT NOT NULL,
  PetId INT NOT NULL,
  ReservationPetID INT NOT NULL,
  PRIMARY KEY (MedicationId),
  FOREIGN KEY (ReservationId, PetId, ReservationPetID) REFERENCES PetReservation(ReservationId, PetId, ReservationPetID)
);

CREATE TABLE PetVaccination
(
  ExpirationDate DATE NOT NULL,
  PetId INT NOT NULL,
  VaccinationId INT NOT NULL,
  PRIMARY KEY (PetId, VaccinationId),
  FOREIGN KEY (PetId) REFERENCES Pet(PetId),
  FOREIGN KEY (VaccinationId) REFERENCES Vaccination(VaccinationId)
);

CREATE TABLE PetReservationService
(
  ServiceId INT NOT NULL,
  ReservationId INT NOT NULL,
  PetId INT NOT NULL,
  ReservationPetID INT NOT NULL,
  PRIMARY KEY (ServiceId),
  FOREIGN KEY (ServiceId) REFERENCES Service(ServiceId),
  FOREIGN KEY (ReservationId, PetId, ReservationPetID) REFERENCES PetReservation(ReservationId, PetId, ReservationPetID)
);

CREATE TABLE Discount
(
  DiscountID INT NOT NULL,
  DiscountRate INT NOT NULL,
  DiscountName VARCHAR(20) NOT NULL,
  PRIMARY KEY (DiscountID)
);
